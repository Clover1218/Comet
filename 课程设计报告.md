
# 基于 Socket 的局域网文件传输系统设计与实现

## 1 引言

### 1.1 项目背景

随着局域网环境中设备数量的增加，用户对高效、可靠的文件传输需求日益增长。传统的文件传输方式（如U盘拷贝、云盘分享）存在传输速度慢、依赖外部网络等问题。本项目旨在设计并实现一个基于 TCP Socket 的局域网文件传输系统，支持断点续传、节点自动发现、并发传输等功能，为用户提供高效可靠的文件共享体验。

### 1.2 项目目标

本系统实现以下核心功能：
- 基于 mDNS 的局域网节点自动发现
- 支持大文件传输和断点续传
- 文件夹压缩传输
- 并发分块传输以提高效率
- 传输进度实时反馈

### 1.3 技术选型

| 分类 | 技术 | 版本 | 选型理由 |
|------|------|------|----------|
| 编程语言 | Go | 1.25 | 高性能并发支持、简洁语法、丰富标准库 |
| 网络协议 | TCP Socket | - | 可靠传输、全双工通信 |
| 服务发现 | mDNS (zeroconf) | - | 无需中心服务器、自动发现 |
| 数据压缩 | Zstd | - | 高压缩率、快速压缩解压 |
| 持久化存储 | 文件系统 | - | 简单可靠、易于部署 |

使用 `errgroup` 实现并发传输：

```go

for chunkIdx := 0; chunkIdx < totalChunks; chunkIdx++ {
    idx := chunkIdx
    if completed[idx] { continue }  // 跳过已完成块
    
    g.Go(func() error {
        // 读取分块、发送、更新状态
        ...
    })

```

**(3) 位图序列化算法**

将 `[]bool` 压缩为字节数组传输：

```go
func boolsToBytes(bits []bool) []byte {
    n := (len(bits) + 7) / 8
    data := make([]byte, 4+n)
    binary.BigEndian.PutUint32(data[:4], uint32(len(bits)))
    for i, v := range bits {
        if v {
            data[4+i/8] |= 1 << (i % 8)
        }
    }
    return data
}
```

func (s *Sender) SendStream(ctx context.Context, path string, targetAddr string, 
    isFolder bool, sessionID string) error {
    
    // 1. 创建连接
    conn, err := s.transport.Dial(ctx, targetAddr)
    defer conn.Close()
    
    // 2. 认证
    s.authenticate(conn)
    
    // 3. 如果是文件夹，先压缩
    if isFolder {
        tmpFilePath := filepath.Join("./data/tmp", sessionID+".tar.zst")
        zipper.NewZstdZipper(0).Pack(path, tmpFilePath)
        finalPath = tmpFilePath
    }
    
    // 4. 发送元数据
    meta := fmt.Sprintf("%s|%d|%s|%d", displayName, totalSize, sessionID, isFolderFlag)
    conn.SendPacket(network.CmdMeta, []byte(meta))
    
    // 5. 断点续传协商
    // 查询对端状态、合并位图、同步状态
    ...
    
    // 6. 并发分块传输
    g, ctx := errgroup.WithContext(ctx)
    g.SetLimit(s.maxWorkers)
    
    for chunkIdx := 0; chunkIdx < totalChunks; chunkIdx++ {
        idx := chunkIdx
        if completed[idx] { continue }
        
        g.Go(func() error {
            // 读取分块
            data := make([]byte, size)
            sharedFile.ReadAt(data, offset)
            
            // 发送分块
            conn.SendPacket(network.CmdChunk, chunkPayload)
            
            // 更新状态
            completed[idx] = true
            s.store.UpdateCheckpoint(sessionID, func(cp *models.Checkpoint) {
                cp.Completed = completed
            })
            return nil
        })
    }
    
    // 7. 发送完成
    conn.SendPacket(network.CmdComplete, []byte(sessionID))
    return nil
}

func (pm *peerManager) Start(ctx context.Context) error {
    // 注册自身服务（mDNS）
    pm.service, err = zeroconf.Register(
        pm.selfID,
        "_comet._tcp.local.",
        "local.",
        pm.port,
        map[string]string{"hostname": pm.selfID},
        nil,
    )
    
    // 启动浏览其他节点
    pm.wg.Add(1)
    go pm.browse(ctx)
    
    return nil
}

func (pm *peerManager) browse(ctx context.Context) {
    entries := make(chan *zeroconf.ServiceEntry)
    go func(results <-chan *zeroconf.ServiceEntry) {
        for entry := range results {
            pm.mu.Lock()
            peer := &models.Peer{
                ID:       entry.HostName,
                Addr:     fmt.Sprintf("%s:%d", entry.AddrIPv4[0], entry.Port),
                Hostname: entry.HostName,
                Online:   true,
                LastSeen: time.Now(),
            }
            pm.peers[peer.ID] = peer
            pm.mu.Unlock()
            
            // 发送事件
            pm.eventCh <- Event{Type: EventPeerJoined, Peer: peer}
        }
    }(entries)
    
    pm.resolver.Browse(ctx, "_comet._tcp.local.", "local.", entries)
}

    ln, err := r.transport.Listen(ctx, listenAddr)
    defer ln.Close()
    
    r.logger.Infof("[Receiver] 监听启动: %s", listenAddr)
    
    for {
        select {

            return ctx.Err()
        default:
        }
        
        conn, err := ln.Accept()
        go r.handleConnection(ctx, conn)  // 并发处理连接

    defer conn.Close()

    r.authenticate(conn)

    cmd, payload, _ := conn.ReadPacket()
    if cmd == network.CmdMeta {
        meta := string(payload)
        parts := strings.SplitN(meta, "|", 4)
        fileName := parts[0]
        totalSize, _ := strconv.ParseInt(parts[1], 10, 64)
        sessionID := parts[2]
        isFolder := parts[3] == "1"
    }
    
    // 3. 断点续传协商
    // 响应查询、接收合并后的位图
    ...
    
    // 4. 接收分块
    for {
        cmd, payload, _ := conn.ReadPacket()
        
        if cmd == network.CmdComplete {
            break
        }
        
        if cmd == network.CmdChunk {
            // 解析分块索引和数据
            idx := int(binary.BigEndian.Uint32(payload[:4]))
            data := payload[8:]
            
            // 写入文件
            f.WriteAt(data, int64(idx)*r.chunkSize)
            completed[idx] = true
        }
    }
    
    // 5. 保存并解压
    os.Rename(tmpFile, savePath)
    if isFolder {
        zipper.NewZstdZipper(0).Unpack(savePath, sessionDir)
        os.Remove(savePath)
    }
}

**(3) 网络传输协议** [internal/network/protocol.go]

```go
func Encode(cmd byte, payload []byte) []byte {
    // [Magic 4B][Cmd 1B][PayloadLen 4B][Payload]
    data := make([]byte, 9+len(payload))
    binary.BigEndian.PutUint32(data[0:4], Magic)
    data[4] = cmd
    binary.BigEndian.PutUint32(data[5:9], uint32(len(payload)))
    copy(data[9:], payload)
    return data
}

func Decode(data []byte) (cmd byte, payload []byte, consumed int, err error) {
    if len(data) < 9 {
        return 0, nil, 0, errors.New("数据不足")
    }
    
    magic := binary.BigEndian.Uint32(data[0:4])
    if magic != Magic {
        return 0, nil, 0, fmt.Errorf("魔数错误: 0x%X", magic)
    }
    
    cmd = data[4]
    payloadLen := binary.BigEndian.Uint32(data[5:9])
    
    if len(data) < 9+int(payloadLen) {
        return 0, nil, 0, errors.New("数据不完整")
    }
    
    payload = data[9 : 9+payloadLen]
    consumed = 9 + int(payloadLen)
    return cmd, payload, consumed, nil
}
```

## 5 总结

### 5.1 完成的功能

本项目成功实现了一个基于 Socket 的局域网文件传输系统，主要功能包括：

1. **节点自动发现**：基于 mDNS 协议实现局域网内节点的自动发现和状态管理
2. **文件传输**：支持文件和文件夹的传输，文件夹自动压缩后传输
3. **断点续传**：使用位图记录传输进度，支持断点恢复
4. **并发传输**：使用 errgroup 实现并发分块传输，提高传输效率
5. **身份认证**：基于 Token 的简单认证机制

### 5.2 技术亮点

1. **位图压缩传输**：将 `[]bool` 压缩为字节数组，减少断点信息传输量
2. **原子文件操作**：使用临时文件 + rename 保证文件写入的原子性
3. **跨平台兼容**：使用标准库，支持 Windows/Linux/macOS
4. **优雅的错误处理**：完善的错误处理和日志记录

### 5.3 改进空间

1. **加密传输**：当前传输未加密，可添加 TLS 支持
2. **文件校验**：可添加 MD5/SHA 校验确保文件完整性
3. **图形界面**：可开发 GUI 版本提升用户体验
4. **限速功能**：可添加传输速率限制
